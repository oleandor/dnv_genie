Command: Set

Description:
Represent a set of modeller objects
Example of use:
Include or remove a selection in a set
MySet = Set();
MySet.add(Bm1);
MySet.add(Bm2);
This command is scripted. Typically when including two beams in the set "MySet"
Put beams in named sets based on the section name
for(var sect in ModelObjects) {
if(sect.supportsType(typeSection)) {
newSet=Set();
for(var bm in ModelObjects) {
if (bm.supportsType(TypeStraightBeam) || bm.supportsType(TypeCurvedBeam)) {
if(bm.section.name==sect.name) {
newSet.add(bm);
}
}
}
newSet.name="Set_"+sect.name;
}
}
or accessing the beams directly
for(var a in ModelObjects)
{
if(a.supportsType(typeStraightBeam))
Print(a.Length);
}

Creating dynamic sets using wild cards
Set1 = DynamicSet(LimitString("Name","Bm2*",false));
// Bm2, Bm20 to B29, Bm200 to 299, and so on will be selected
//Using regexp (regular expressions )
Set2 = DynamicSet(LimitString("Name","Bm2..",true));
Set3 = DynamicSet(LimitString("Name","B[a-z][0-9][0-9]",true));
// Bm 200 to Bm299 will be selected (. Replaces any character )
// Ba00 to Bz99 will be selected
//for more info about regular expression see:
http://en.wikipedia.org/wiki/Regular_expression
Create a new named set containing all beams having at least one end inside MyBoundingBox
Point1 = Point(0,0,0);
Point2 = Point(10,10,10);
MyBoundingBox=limitBox(Point1, Point2);
setOfBeamsInside = Set();
for (var object in ModelObjects){
if ( object.supportsType(typeStraightBeam) || object.supportsType(typeCurvedBeam) ) {
if ( MyBoundingBox.test(object.end1()) || MyBoundingBox.test(object.end2()) ) {
setOfBeamsInside.add(object);
} } }
This script makes a new named set containing all beams having at
least one end inside MyBoundingBox
Delete named set
var arrayOfObjectsInSet = new Array();
for (var namedSetObject in ModelObjects){
//Deleting all objects in "MySet"
if(namedSetObject.supportsType(typeSet) && namedSetObject.name()=="MySet"){
for (var objectInSet in namedSetObject){
arrayOfObjectsInSet[arrayOfObjectsInSet.length] = objectInSet;
} } }
for (i=0;i<=arrayOfObjectsInSet.length-1;i++){
objectToDelete = arrayOfObjectsInSet[i];
delete(objectToDelete);
}
// replace the name of the set with your own
delete(MySet);
//Deleting the set "MySet". If you just use delete(MySet) alone, it will delete the set
//but none of the objects in the set
Merging the two existing names sets into a new named set
var arrayOfObjectsInSet = new Array();
for (var namedSetObject in ModelObjects){
// replace the names of the sets with your own
if( namedSetObject.supportsType(typeSet) && (namedSetObject.name()=="SubSet1" || namedSetObject.name()=="SubSet2") ) {
for (var objectInSet in namedSetObject){
arrayOfObjectsInSet[arrayOfObjectsInSet.length] = objectInSet;
} } }
MainSet = Set();
for (i=0;i<=arrayOfObjectsInSet.length-1;i++){
objectToAdd = arrayOfObjectsInSet[i];
MainSet.add(objectToAdd);
}
